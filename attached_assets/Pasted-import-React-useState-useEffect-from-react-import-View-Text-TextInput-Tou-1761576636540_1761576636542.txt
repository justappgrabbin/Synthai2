// Simple LlamaFile Client for React Native
class LlamaFileClient {
  constructor(baseUrl = 'http://localhost:8080') {
    this.baseUrl = baseUrl;
    this.isConnected = false;
  }

  async checkConnection() {
    try {
      const response = await fetch(`${this.baseUrl}/health`, {
        method: 'GET',
        timeout: 3000
      });
      this.isConnected = response.ok;
      return this.isConnected;
    } catch (error) {
      this.isConnected = false;
      return false;
    }
  }

  async generateCode(prompt) {
    try {
      const response = await fetch(`${this.baseUrl}/v1/chat/completions`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          messages: [
            {
              role: "system", 
              content: "You are a React Native expert. Generate clean, working React Native code with proper imports and styling. Always include StyleSheet."
            },
            {
              role: "user", 
              content: `Generate React Native code for: ${prompt}`
            }
          ],
          temperature: 0.3,
          max_tokens: 1500
        })
      });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const data = await response.json();
      return data.choices[0].message.content;
    } catch (error) {
      console.error('LlamaFile API error:', error);
      throw error;
    }
  }
}

// Connection Status Component
const ConnectionStatus = ({ isConnected, onCheck }) => {
  const [checking, setChecking] = useState(false);

  const handleCheck = async () => {
    setChecking(true);
    await onCheck();
    setChecking(false);
  };

  return (
    <View style={styles.statusContainer}>
      <View style={styles.statusRow}>
        <Text style={styles.statusLabel}>LlamaFile Status:</Text>
        <View style={[
          styles.statusBadge, 
          { backgroundColor: isConnected ? '#10B981' : '#EF4444' }
        ]}>
          <Text style={styles.statusText}>
            {isConnected ? 'Connected' : 'Disconnected'}
          </Text>
        </View>
      </View>
      
      <TouchableOpacity 
        style={styles.checkButton} 
        onPress={handleCheck}
        disabled={checking}
      >
        {checking ? (
          <ActivityIndicator color="white" size="small" />
        ) : (
          <Text style={styles.checkButtonText}>Test Connection</Text>
        )}
      </TouchableOpacity>

      {!isConnected && (
        <View style={styles.setupInfo}>
          <Text style={styles.setupTitle}>Setup Required:</Text>
          <Text style={styles.setupText}>1. Download llamafile</Text>
          <Text style={styles.setupText}>2. Run: ./llamafile --server --port 8080</Text>
          <Text style={styles.setupText}>3. Test connection above</Text>
        </View>
      )}
    </View>
  );
};

// Code Generator Component
const CodeGenerator = ({ client, isConnected }) => {
  const [prompt, setPrompt] = useState('');
  const [generatedCode, setGeneratedCode] = useState('');
  const [isGenerating, setIsGenerating] = useState(false);

  const generateCode = async () => {
    if (!isConnected || !prompt.trim()) {
      Alert.alert('Error', 'Please ensure connection and enter a prompt');
      return;
    }

    setIsGenerating(true);
    try {
      const code = await client.generateCode(prompt);
      setGeneratedCode(code);
    } catch (error) {
      Alert.alert('Error', `Generation failed: ${error.message}`);
    } finally {
      setIsGenerating(false);
    }
  };

  const samplePrompts = [
    "login screen with email and password",
    "button component with different variants",
    "user profile screen with avatar",
    "list item with title and subtitle"
  ];

  return (
    <View style={styles.generatorContainer}>
      <Text style={styles.sectionTitle}>AI Code Generator</Text>
      
      {/* Sample Prompts */}
      <Text style={styles.samplesTitle}>Quick Start:</Text>
      <ScrollView horizontal showsHorizontalScrollIndicator={false} style={styles.samplesContainer}>
        {samplePrompts.map((sample, index) => (
          <TouchableOpacity
            key={index}
            style={styles.sampleChip}
            onPress={() => setPrompt(sample)}
          >
            <Text style={styles.sampleText}>{sample}</Text>
          </TouchableOpacity>
        ))}
      </ScrollView>

      {/* Prompt Input */}
      <TextInput
        style={styles.promptInput}
        placeholder="Describe the React Native component you want..."
        value={prompt}
        onChangeText={setPrompt}
        multiline
        editable={isConnected}
      />

      {/* Generate Button */}
      <TouchableOpacity 
        style={[
          styles.generateButton,
          (!isConnected || !prompt.trim() || isGenerating) && styles.generateButtonDisabled
        ]}
        onPress={generateCode}
        disabled={!isConnected || !prompt.trim() || isGenerating}
      >
        {isGenerating ? (
          <View style={styles.generatingContainer}>
            <ActivityIndicator color="white" size="small" />
            <Text style={styles.generateButtonText}>Generating...</Text>
          </View>
        ) : (
          <Text style={styles.generateButtonText}>Generate Code</Text>
        )}
      </TouchableOpacity>

      {/* Generated Code Display */}
      {generatedCode && (
        <View style={styles.codeContainer}>
          <Text style={styles.codeTitle}>Generated Code:</Text>
          <ScrollView style={styles.codeScroll}>
            <Text style={styles.codeText}>{generatedCode}</Text>
          </ScrollView>
        </View>
      )}
    </View>
  );
};

// Main App Component
export default function LlamaFileApp() {
  const [client] = useState(() => new LlamaFileClient());
  const [isConnected, setIsConnected] = useState(false);

  const checkConnection = async () => {
    const connected = await client.checkConnection();
    setIsConnected(connected);
  };

  useEffect(() => {
    checkConnection();
  }, []);

  return (
    <View style={styles.container}>
      <ScrollView contentContainerStyle={styles.scrollContent}>
        {/* Header */}
        <View style={styles.header}>
          <Text style={styles.title}>LlamaFile Code Generator</Text>
          <Text style={styles.subtitle}>Local AI for React Native Development</Text>
        </View>

        {/* Connection Status */}
        <ConnectionStatus 
          isConnected={isConnected} 
          onCheck={checkConnection} 
        />

        {/* Code Generator */}
        <CodeGenerator 
          client={client} 
          isConnected={isConnected} 
        />
      </ScrollView>
    </View>
  );
}
