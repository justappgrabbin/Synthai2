  <script>
    // DIMENSIONAL ARCHITECTURE
    const DIMENSIONS = {
      Movement: {
        symbol: '⟡',
        nature: 'Energy → Creation → Seeing',
        component: 'Magnetic Monopole',
        keynote: 'I Define',
        field: 'Spirit',
        color: 'from-red-500 to-orange-500'
      },
      Evolution: {
        symbol: '⟁',
        nature: 'Gravity → Memory → Taste',
        component: 'Personality Crystal',
        keynote: 'I Remember',
        field: 'Mind',
        color: 'from-purple-500 to-pink-500'
      },
      Being: {
        symbol: '⬢',
        nature: 'Matter → Touch → Sex',
        component: 'The Atom',
        keynote: 'I Am',
        field: 'Body',
        color: 'from-green-500 to-emerald-500'
      },
      Design: {
        symbol: '◇',
        nature: 'Structure → Progress → Smell',
        component: 'Design Crystal',
        keynote: 'I Design',
        field: 'Soul',
        color: 'from-blue-500 to-cyan-500'
      },
      Space: {
        symbol: '◉',
        nature: 'Form → Illusion → Hearing',
        component: 'Personality Crystal',
        keynote: 'I Think',
        field: 'Heart',
        color: 'from-yellow-500 to-amber-500'
      }
    };

    let selectedDimension = 'Evolution';

    // Initialize dimension cards
    function initDimensionCards() {
      const container = document.getElementById('dimensionCards');
      container.innerHTML = Object.entries(DIMENSIONS).map(([name, data]) => `
        <div 
          class="dimension-card border border-white/10 rounded-lg p-4 ${name === selectedDimension ? 'selected' : ''}"
          onclick="selectDimension('${name}')"
        >
          <div class="flex items-center justify-between mb-2">
            <div class="flex items-center gap-3">
              <span class="text-3xl">${data.symbol}</span>
              <div>
                <div class="font-bold">${name}</div>
                <div class="text-xs text-gray-400">${data.field} · ${data.keynote}</div>
              </div>
            </div>
          </div>
          <div class="text-xs text-gray-500 mt-2">${data.nature}</div>
          <div class="text-xs text-gray-600 mt-1">${data.component}</div>
        </div>
      `).join('');
    }

    function selectDimension(name) {
      selectedDimension = name;
      initDimensionCards();
      updateGlyph();
    }

    // Update field value displays
    ['mind', 'body', 'heart', 'spirit', 'soul'].forEach(field => {
      document.getElementById(field).addEventListener('input', (e) => {
        document.getElementById(field + 'Value').textContent = e.target.value + '%';
        updateGlyph();
      });
    });

    document.getElementById('coherence').addEventListener('input', (e) => {
      document.getElementById('coherenceValue').textContent = e.target.value + '%';
      updateGlyph();
    });

    // Update glyph on any input change
    ['moduleName', 'repo', 'path', 'connections', 'state', 'personality', 'design', 'monopole'].forEach(id => {
      document.getElementById(id).addEventListener('input', updateGlyph);
      document.getElementById(id).addEventListener('change', updateGlyph);
    });

    function generateID() {
      const dim = DIMENSIONS[selectedDimension];
      const keynoteWord = dim.keynote.split(' ')[1].toUpperCase();
      const counter = String(Math.floor(Math.random() * 999)).padStart(3, '0');
      return `${dim.symbol}-${keynoteWord}-${counter}`;
    }

    function updateGlyph() {
      const name = document.getElementById('moduleName').value || 'Unnamed Module';
      const repo = document.getElementById('repo').value || 'unknown';
      const path = document.getElementById('path').value || '/';
      const connections = document.getElementById('connections').value
        .split(',')
        .map(c => c.trim())
        .filter(c => c);
      
      const personality = document.getElementById('personality').checked;
      const design = document.getElementById('design').checked;
      const monopole = document.getElementById('monopole').checked;

      const fieldResonance = {
        mind: parseInt(document.getElementById('mind').value) / 100,
        body: parseInt(document.getElementById('body').value) / 100,
        heart: parseInt(document.getElementById('heart').value) / 100,
        spirit: parseInt(document.getElementById('spirit').value) / 100,
        soul: parseInt(document.getElementById('soul').value) / 100
      };

      const coherence = parseInt(document.getElementById('coherence').value) / 100;
      const state = document.getElementById('state').value;

      const dim = DIMENSIONS[selectedDimension];
      const id = generateID();

      const stateEmoji = {
        active: '⚡',
        prototype: '🔬',
        deprecated: '⏸️',
        broken: '⚠️'
      }[state];

      const coherenceBar = '◉'.repeat(Math.floor(coherence * 10)) + '○'.repeat(10 - Math.floor(coherence * 10));

      // Generate visual glyph
      const visual = `
╭─────────────────────────────────╮
│   ${personality ? '▲' : '△'} ${name.toUpperCase().padEnd(25).slice(0, 25)} │
│  ╱ ╲         ${id.padEnd(17).slice(0, 17)} │
│ ╱   ╲                           │
│╱─────╲   ${dim.component.padEnd(21).slice(0, 21)} │
││ ${dim.symbol}  ││   ${(dim.field + ' Field').padEnd(21).slice(0, 21)} │
│╲     ╱                           │
│ ╲   ╱    ${dim.keynote.padEnd(21).slice(0, 21)} │
│  ╲ ╱     ${dim.nature.split('→')[0].trim().padEnd(21).slice(0, 21)} │
│   ${design ? '▼' : '▽'}      ${monopole ? '◆' : '◇'} Monopole              │
│                                 │
│  Field Resonance:               │
│  Mind   ${'█'.repeat(Math.floor(fieldResonance.mind * 10))}${'░'.repeat(10 - Math.floor(fieldResonance.mind * 10))} ${Math.floor(fieldResonance.mind * 100)}%   │
│  Soul   ${'█'.repeat(Math.floor(fieldResonance.soul * 10))}${'░'.repeat(10 - Math.floor(fieldResonance.soul * 10))} ${Math.floor(fieldResonance.soul * 100)}%   │
│  Heart  ${'█'.repeat(Math.floor(fieldResonance.heart * 10))}${'░'.repeat(10 - Math.floor(fieldResonance.heart * 10))} ${Math.floor(fieldResonance.heart * 100)}%   │
│  Spirit ${'█'.repeat(Math.floor(fieldResonance.spirit * 10))}${'░'.repeat(10 - Math.floor(fieldResonance.spirit * 10))} ${Math.floor(fieldResonance.spirit * 100)}%   │
│  Body   ${'█'.repeat(Math.floor(fieldResonance.body * 10))}${'░'.repeat(10 - Math.floor(fieldResonance.body * 10))} ${Math.floor(fieldResonance.body * 100)}%   │
│                                 │
${connections.length > 0 ? `│  Connections:                   │\n${connections.map(c => `│  → ${c.padEnd(29).slice(0, 29)} │`).join('\n')}\n│                                 │` : ''}
│  ${stateEmoji} ${state.toUpperCase()} | Coherence: ${Math.floor(coherence * 100)}% ${coherenceBar} │
│                                 │
│  Repo: ${repo.padEnd(24).slice(0, 24)} │
│  Path: ${path.padEnd(24).slice(0, 24)} │
╰─────────────────────────────────╯`;

      // Generate file header
      const header = `/**
 * ╔═══════════════════════════════════════╗
 * ║  ${personality ? '▲' : '△'} ${name.toUpperCase().padEnd(33).slice(0, 33)} ║
 * ║ ╱ ╲    ${id.padEnd(30).slice(0, 30)} ║
 * ║╱───╲   ${(dim.field + ' · ' + dim.component).padEnd(30).slice(0, 30)} ║
 * ║│ ${dim.symbol} │   ${dim.nature.split('→').slice(0, 2).join(' → ').padEnd(30).slice(0, 30)} ║
 * ║╲───╱   ${dim.keynote.padEnd(30).slice(0, 30)} ║
 * ║  ${design ? '▼' : '▽'}                                   ║
 * ╠═══════════════════════════════════════╣
 * ║ Repository: ${repo.padEnd(26).slice(0, 26)} ║
 * ║ Path: ${path.padEnd(32).slice(0, 32)} ║
 * ║ State: ${state.toUpperCase().padEnd(31).slice(0, 31)} ║
 * ║ Coherence: ${Math.floor(coherence * 100)}%${' '.repeat(27)} ║
 * ╚═══════════════════════════════════════╝
 */`;

      // Generate JSON
      const json = {
        glyph_id: id,
        name: name,
        symbol: dim.symbol,
        dimension: {
          name: selectedDimension,
          symbol: dim.symbol,
          keynote: dim.keynote,
          nature_chain: dim.nature.split(' → '),
          component: dim.component,
          field: dim.field
        },
        crystal_alignment: { personality, design, monopole },
        field_resonance: fieldResonance,
        connections: connections,
        coherence: coherence,
        state: state,
        repo: repo,
        path: path,
        generated_at: new Date().toISOString()
      };

      document.getElementById('glyphPreview').textContent = visual;
      document.getElementById('headerPreview').textContent = header;
      document.getElementById('jsonPreview').textContent = JSON.stringify(json, null, 2);

      // Store for export
      window.currentGlyph = { visual, header, json };
    }

    function copyHeader() {
      navigator.clipboard.writeText(window.currentGlyph.header);
      alert('✨ File header copied to clipboard!');
    }

    function downloadJSON() {
      const blob = new Blob([JSON.stringify(window.currentGlyph.json, null, 2)], { type: 'application/json' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `glyph-${window.currentGlyph.json.glyph_id}.json`;
      a.click();
    }

    // Initialize
    initDimensionCards();
    updateGlyph();
  </script>
